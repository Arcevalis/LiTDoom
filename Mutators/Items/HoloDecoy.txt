ACTOR HoloDecoy
{
	MONSTER
	-COUNTKILL
	+NOBLOOD
	Radius 10
	Height 9
	Health 1500
	Mass 0x7FFFFFFF
	DamageFactor 0.0
	Scale 0.5
	Speed 2
	States
	{
	Spawn:
		HOLP ABCD 3
		HOLP ABCD 2
		TNT1 A 0 A_SpawnItemEx("HoloDecoyProjection",0,0,10,0,0,0,0,SXF_SETMASTER)
		HOLP ABCD 1
	  SpawnLoop:
		HOLP ABCD 1 A_Wander
		TNT1 A 0 A_DamageSelf(10,"None",DMSS_NOFACTOR)
		Loop
	Death:
		HOLP ABCD 2 A_Wander
		HOLP ABCD 3 A_Wander
		HOLP ABCD 4 A_Wander
		HOLP ABCD 5 A_Wander
		TNT1 A 0 A_KillChildren("HoloKill")
		HOLP A 1 A_FadeOut(0.1,1)
		Wait
	}
}

ACTOR HoloDecoyProjection
{
	MONSTER
	+FRIENDLY
	-COUNTKILL
	+GHOST
	+NOBLOODDECALS
	+DONTFALL
	+NOGRAVITY
	BloodType ""
	Height 45
	Radius 5
	Health 1
	Mass 0x7FFFFFFF
	DamageFactor 0.0
	DamageFactor "HoloKill",1
	RenderStyle "Add"
	Alpha 0.75
	Scale 0.75
	States
	{
	Spawn:
		TNT1 A 0 NoDelay Thing_ChangeTID(0,666)
	 SpawnLoop:
		HOLO A 0 A_RadiusGive("DecoyHater",512,RGF_MONSTERS)
		HOLO AAAABBBBCCCCDDDD 1 BRIGHT A_Warp(AAPTR_MASTER,0,0,15,0,WARPF_NOCHECKPOSITION|WARPF_INTERPOLATE)
		Loop
	Death.HoloKill:
		"####" "#" 0 A_NoBlocking
	 FadeLoop:
		"####" "#" 0 BRIGHT A_SetScale(scalex-0.075,scaley+0.075)
		"####" "#" 2 BRIGHT A_FadeOut(0.1,1)
		Loop
	}
}

ACTOR DecoyHater : CustomInventory
{
	+ALWAYSPICKUP
	States
	{
	Pickup:
		TNT1 A 0 A_JumpIf(CheckClass("HoloDecoy"),"End")

		TNT1 A 0 A_JumpIf(CheckClass("Recombinerstatic"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("RecombinerStart"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("Sentrybot"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("RECOMBINERRIP"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("riflemarineguarding"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("shotgunmarineguarding"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("plasmagunmarineguarding"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("minigunmarineguarding"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("Marine_rifle"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("Marine_shotgun"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("Marine_minigun"),"End")
		TNT1 A 0 A_JumpIf(CheckClass("Marine_plasmagun"),"End")
		
		TNT1 A 0 A_JumpIf(tidtohate==666,"End")
		TNT1 A 0 Thing_Hate(0,666,0)
	 End:
		TNT1 A 0
		Stop
	}
}
